307 # than I've already had to...).

310     def send_request_with_body_stream(sock, ver, path, f)
 
311       raise ArgumentError, "Content-Length not given and Transfer-Encoding is not `chunked'" unless content_length() or chunked?

314         set_content_type 'application/x-www-form-urlencoded'
 
315       end
