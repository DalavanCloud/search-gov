464   def shorten_url (url, length=30)
 
465     return url if url.length <= length
 
466     if url.count('/') >= 4
 
467       if url.index('?')
 
468         arr = url[0..url.index('?')].split('/')
 
469         arr[arr.size - 1] += url[url.index('?') + 1..-1]
 
470       else
 
471         arr = url.split('/')
 
472       end
 
473       host = arr[0]+"//"+arr[2]
 
474       if arr.last.index('?')
 
475         doc_path = arr.last.split('?')
 
476         doc = [doc_path.first, "?", doc_path[1].split('&').first, "..."].join
 
477       else
 
478         doc = arr.last.split('?').first
 
479       end
 
480       [host, "...", doc].join('/')
 
481     else
 
482       url[0, length]+"..."
 
483     end
 
484   end
