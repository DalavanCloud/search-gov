123   def normalize_domains(staged = true)
 
124     method = staged ? "staged_domains" : "domains"
 
125     return if self.send(method).nil?
 
126     domain_list = self.send(method).gsub(/(https?:\/\/| )/, '').split.
 
127       select { |domain| domain =~ /^[a-z0-9]+([\-\.]{1}[a-z0-9]+)*\.[a-z]{2,3}(\/.*)?$/ix }.
 
128       sort { |a, b| a.length <=> b.length }.uniq
 
129     result = []
 
130     while (domain_list.length > 0)
 
131       result << domain_list.first
 
132       domain_list = domain_list.drop(1).delete_if { |domain| domain.start_with?(domain_list.first) or domain.include?(".#{domain_list.first}") }
 
133     end
 
134     self.send(method + "=", result.join("\n"))
