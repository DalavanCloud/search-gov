320   def self.process_cpsc_row(row)
 
321     recall = Recall.find_or_initialize_by_recall_number(:recall_number => row[0], :y2k => row[1], :organization => 'CPSC')
 
322     recall.recalled_on ||= Date.parse(row[9]) rescue nil
 
323     CPSC_FULL_TEXT_SEARCH_FIELDS.each_pair do |detail_type, column_index|
 
324       conditions = ['detail_type = ? AND detail_value = ?', detail_type, row[column_index]]
 
325       unless row[column_index].blank? or (!recall.new_record? && recall.recall_details.exists?(conditions))
 
326         recall.recall_details << RecallDetail.new(:detail_type => detail_type, :detail_value => row[column_index])
 
327       end
 
328     end
 
329     recall.save!
 
330   end
